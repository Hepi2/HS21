2.1.1 Assoziation zwei kategorialer Variabeln (Haarfarbe vs. Augenfarbe)

2.1.1.1 Chi-Quadrat-Test

haar_augen <- data.frame( 
  row.names = c("Helle Haare", "Dunkle Haare"),
  Blaue_Augen = c(38,11),
  Braune_Augen = c(14,51)
  )

haar_augen
##              Blaue_Augen Braune_Augen
## Helle Haare           38           14
## Dunkle Haare          11           51
qchisq(0.95,1)
## [1] 3.841459
count <- as.matrix(haar_augen)

chisq.test(count)
## 
##  Pearson's Chi-squared test with Yates' continuity correction
## 
## data:  count
## X-squared = 33.112, df = 1, p-value = 8.7e-09
